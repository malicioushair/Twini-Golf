cmake_minimum_required(VERSION 3.19.0)
project(Twini-Golf CXX)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

if(NOT EXISTS "${CMAKE_BINARY_DIR}/conan.cmake")
     message(STATUS "Downloading conan.cmake from https://github.com/conan-io/cmake-conan")
     file(DOWNLOAD "https://raw.githubusercontent.com/conan-io/cmake-conan/master/conan.cmake"
     "${CMAKE_BINARY_DIR}/conan.cmake")
endif()

list(APPEND CMAKE_MODULE_PATH ${CMAKE_CURRENT_SOURCE_DIR}/cmake/sdl2)
find_package(SDL2 REQUIRED)

include(${CMAKE_BINARY_DIR}/conan.cmake)
conan_cmake_run(
     CONANFILE conanfile.txt
     BASIC_SETUP
     BUILD missing
)

file(GLOB FONTS "res/font/*.ttf")
file(GLOB GFX "res/gfx/*.png")
file(GLOB SFX "res/sfx/*.mp3")

file(COPY ${FONTS} DESTINATION ${CMAKE_CURRENT_BINARY_DIR}/bin/res/font)
file(COPY ${GFX} DESTINATION ${CMAKE_CURRENT_BINARY_DIR}/bin/res/gfx)
file(COPY ${SFX} DESTINATION ${CMAKE_CURRENT_BINARY_DIR}/bin/res/sfx)

file(GLOB SRC_FILES
     "src/*.h"
     "src/*.cpp"
)

find_file(SDL2_INCLUDE_DIR NAME SDL.h HINTS SDL2)
find_library(SDL2_LIBRARY NAME SDL2)

add_executable(${PROJECT_NAME} ${SRC_FILES})
target_link_directories(${PROJECT_NAME} PUBLIC ${CMAKE_CURRENT_BINARY_DIR}/bin)

include(${CMAKE_BINARY_DIR}/conanbuildinfo.cmake)
target_link_libraries(${PROJECT_NAME} ${CONAN_LIBS})

set_property(TARGET ${PROJECT_NAME} PROPERTY VS_DEBUGGER_WORKING_DIRECTORY "${CMAKE_CURRENT_BINARY_DIR}/bin")
